== 조건별로 분류하여 주문상태 출력하기 ==


select order_id, product_id, out_date,
       case when to_char(out_date, 'yyyy-mm-dd') <= '2022-05-01'  then '출고완료'
            WHEN to_char(out_date, 'yyyy-mm-dd') > '2022-05-01'  then '출고대기'
       else '출고미정'
       end AS "출고여부"
from food_order
Order by order_id;
